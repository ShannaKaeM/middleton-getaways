WordPress Block Pattern Example

How to Use in WordPress:

<!-- wp:generateblocks/container {"className":"mi-card card-style-2","htmlAttributes":{"data-card-style":"2"}} -->
<div class="wp-block-generateblocks-container mi-card card-style-2" data-card-style="2">
    
    <!-- wp:heading {"level":3} -->
    <h3>Premium Property</h3>
    <!-- /wp:heading -->
    
    <!-- wp:paragraph -->
    <p>Experience luxury in our premium vacation rentals.</p>
    <!-- /wp:paragraph -->
    
    <!-- wp:generateblocks/button {"className":"mi-btn btn-primary","htmlAttributes":{"data-btn":"primary"}} -->
    <a class="wp-block-generateblocks-button mi-btn btn-primary" data-btn="primary">Book Now</a>
    <!-- /wp:generateblocks/button -->
    
</div>
<!-- /wp:generateblocks/container -->
